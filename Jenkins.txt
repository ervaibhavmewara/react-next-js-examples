    agent any
    stages{
        stage("build") {
            steps {
                 sh " yarn install"
                 sh " yarn build" 
            }
        }    
        stage("deploy") {
            steps {
                   sh "pm2 restart react-next || pm2 start --yarn name react-next --start -p 8080"
                  

            } 
       }    
    }

}
